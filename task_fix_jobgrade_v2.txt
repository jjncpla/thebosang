## 세션 타입: 버그픽스 세션
## 작업 목표
Contact jobGrade가 전부 기타로 분류된 원인 진단 및 수정.

---

## Step 1: 진단 API 생성

파일: `app/api/debug-contacts/route.ts`

```typescript
import { NextResponse } from 'next/server'
import { prisma } from '@/lib/prisma'

export async function GET() {
  try {
    // 실제 title 값 목록 확인
    const titles = await prisma.contact.groupBy({
      by: ['title'],
      _count: { id: true },
      orderBy: { _count: { id: 'desc' } },
      take: 50,
    })

    // jobGrade 현황
    const grades = await prisma.contact.groupBy({
      by: ['jobGrade'],
      _count: { id: true },
    })

    // 샘플 데이터 5개
    const samples = await prisma.contact.findMany({
      take: 5,
      select: { name: true, title: true, jobGrade: true }
    })

    return NextResponse.json({ titles, grades, samples })
  } catch(e: any) {
    return NextResponse.json({ error: e.message }, { status: 500 })
  }
}
```

middleware.ts에서 `/api/debug-contacts` 인증 제외 처리.

## Step 3: title 값 확인 후 fix-job-grade API 수정

debug-contacts 결과를 보고 실제 title 값에 맞게
fix-job-grade API의 contains 조건을 수정한다.

예를 들어 title이 "이사/공인노무사" 형태라면
`{ contains: '노무사' }` 로 잡히므로 순서 조정 필요.

실제 title 값 확인 후 아래 로직으로 fix-job-grade API를 재작성한다:

파일: `app/api/fix-job-grade/route.ts`

```typescript
import { NextResponse } from 'next/server'
import { prisma } from '@/lib/prisma'

// title 문자열로 jobGrade 판별
function classify(title: string): string {
  if (!title) return '기타'
  // 등기노무사 먼저 (더 구체적인 조건)
  if (/수석노무사|지사장|권역지사장|권역장|센터장|부센터장|부대표|전무이사|부문장/.test(title)) return '등기노무사'
  if (/^대표$/.test(title.trim())) return '등기노무사'
  if (/이사/.test(title) && !/전무이사/.test(title)) return '등기노무사'
  // 변호사
  if (/변호사/.test(title)) return '변호사'
  // 노무사 (공인노무사, 전임노무사, 수습노무사 등 포함)
  if (/노무사/.test(title)) return '노무사'
  // 외근직
  if (/팀장|차장|과장/.test(title)) return '외근직'
  // 내근직
  if (/파트장|매니저|실장|부장|사원|주임|대리|디자이너|계장/.test(title)) return '내근직'
  return '기타'
}

export async function GET() {
  try {
    // 전체 contact 가져와서 JS에서 분류 후 개별 업데이트
    const all = await prisma.contact.findMany({
      select: { id: true, title: true }
    })

    let updated = 0
    // 배치로 업데이트
    for (const c of all) {
      const grade = classify(c.title)
      await prisma.contact.update({
        where: { id: c.id },
        data: { jobGrade: grade }
      })
      updated++
    }

    // 결과 집계
    const stats = await prisma.contact.groupBy({
      by: ['jobGrade'],
      _count: { id: true },
      orderBy: { _count: { id: 'desc' } }
    })

    return NextResponse.json({
      ok: true,
      total: updated,
      stats: stats.map(s => ({ grade: s.jobGrade, count: s._count.id }))
    })
  } catch(e: any) {
    return NextResponse.json({ ok: false, error: e.message }, { status: 500 })
  }
}
```

middleware.ts에서 `/api/fix-job-grade` 인증 제외 처리 (이미 되어 있으면 생략).
