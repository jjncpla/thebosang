```typescript
async function handleCaseQuery(chatId: number, searchName: string) {
  if (!searchName) {
    await sendTelegramMessage(chatId, '조회할 재해자 성명을 입력해주세요.\n예: /조회 홍길동')
    return
  }

  const cases = await prisma.case.findMany({
    where: { patient: { name: { contains: searchName } } },
    include: { patient: { select: { name: true } } },
    take: 5,
  })

  if (cases.length === 0) {
    await sendTelegramMessage(chatId, `"${searchName}" 검색 결과가 없습니다.`)
    return
  }

  const resultText = cases.map(c => {
    const statusMap: Record<string, string> = {
      CONSULTING: '상담 중',
      CONTRACTED: '약정 완료',
      SUBMITTED: '공단 접수 완료',
      IN_EXAM: '특진 진행 중',
      EXAM_DONE: '특진 완료',
      APPROVED: '승인',
      REJECTED: '불승인',
      CLOSED: '종결',
      OBJECTION: '이의제기 진행 중',
    }
    const status = statusMap[c.status] ?? c.status
    return `📋 ${c.patient?.name} — ${status}`
  }).join('\n')

  await sendTelegramMessage(chatId, `검색 결과 (${cases.length}건):\n${resultText}`)
}
```

```typescript
function detectKeyword(text: string): KeywordMatch | null {
  // 패턴 1: 승인/불승인 결과 공유
  if (text.includes('승인') && (text.includes('난청') || text.includes('소음'))) {
    return { type: 'APPROVAL_RESULT', value: text }
  }
  // 패턴 2: 특진 일정
  if (text.includes('특진') && text.includes('일정')) {
    return { type: 'EXAM_SCHEDULE', value: text }
  }
  // 패턴 3: 접수 완료
  if (text.includes('접수') && text.includes('완료')) {
    return { type: 'SUBMISSION_DONE', value: text }

  return null
}
```

```typescript
async function sendTelegramMessage(chatId: number, text: string) {
  const token = process.env.TELEGRAM_BOT_TOKEN
  if (!token) return

  await fetch(`https://api.telegram.org/bot${token}/sendMessage`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      chat_id: chatId,
      text,
      parse_mode: 'HTML',
    }),
  })
}
```

## STEP 2: TBSS → 텔레그램 알림 발송 API

파일: app/api/telegram/notify/route.ts (없으면 신규 생성, 있으면 수정)

```typescript
import { NextRequest, NextResponse } from 'next/server'
import { auth } from '@/auth'
import { prisma } from '@/lib/prisma'

export async function POST(req: NextRequest) {
  const session = await auth()
  if (!session) return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })

  const { chatId, message } = await req.json()

  const token = process.env.TELEGRAM_BOT_TOKEN
  if (!token) {
    return NextResponse.json({ error: 'TELEGRAM_BOT_TOKEN 미설정' }, { status: 500 })
  }

  const res = await fetch(`https://api.telegram.org/bot${token}/sendMessage`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      chat_id: chatId,
      text: message,
      parse_mode: 'HTML',
    }),
  })

  const data = await res.json()
  return NextResponse.json(data)
}
```

## STEP 3: 기일/Todo 알림 자동 발송 크론 API

파일: app/api/telegram/cron-notify/route.ts (신규 생성)

```typescript
import { NextRequest, NextResponse } from 'next/server'
import { prisma } from '@/lib/prisma'

// Railway 크론잡 또는 외부 크론으로 매일 오전 9시 호출
// 호출 방법: GET https://thebosang-production.up.railway.app/api/telegram/cron-notify
//            헤더: Authorization: Bearer {CRON_SECRET}

export async function GET(req: NextRequest) {
  // 크론 시크릿 검증
  const auth = req.headers.get('Authorization')
  if (auth !== `Bearer ${process.env.CRON_SECRET}`) {
    return NextResponse.json({ error: 'Forbidden' }, { status: 403 })
  }

  const token = process.env.TELEGRAM_BOT_TOKEN
  if (!token) return NextResponse.json({ error: 'BOT_TOKEN 없음' }, { status: 500 })

  const today = new Date()
  today.setHours(0, 0, 0, 0)
  const tomorrow = new Date(today)
  tomorrow.setDate(tomorrow.getDate() + 1)

  // 오늘 기한인 Todo 조회
  const dueTodos = await prisma.todo.findMany({
    where: {
      isDone: false,
      dueDate: { gte: today, lt: tomorrow },
    },
    include: {
      user: { select: { name: true } },
    },
  })

  if (dueTodos.length === 0) {
    return NextResponse.json({ ok: true, sent: 0 })
  }

  // 담당자별 그룹핑
  const grouped: Record<string, typeof dueTodos> = {}
  for (const todo of dueTodos) {
    const key = todo.assignedTo ?? 'unknown'
    if (!grouped[key]) grouped[key] = []
    grouped[key].push(todo)
  }

  // TBSS_NOTIFY_CHAT_ID 환경변수로 공통 채널에 발송
  const chatId = process.env.TBSS_NOTIFY_CHAT_ID
  if (!chatId) return NextResponse.json({ error: 'CHAT_ID 없음' }, { status: 500 })

  const messageLines = [`📅 <b>오늘의 TBSS 할일 (${dueTodos.length}건)</b>\n`]
  for (const todo of dueTodos) {
    messageLines.push(`• ${todo.title} (${todo.user?.name ?? '담당자 미지정'})`)
  }

  await fetch(`https://api.telegram.org/bot${token}/sendMessage`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      chat_id: chatId,
      text: messageLines.join('\n'),
      parse_mode: 'HTML',
    }),
  })

  return NextResponse.json({ ok: true, sent: dueTodos.length })
}
```
