## STEP 2: app/api/cases/[caseId]/forms/route.ts — NOISE_WORK_CONFIRM 케이스 교체

case 'NOISE_WORK_CONFIRM': {
  const pdfDoc = await loadBlankForm('noise_work_confirm.pdf')
  const font = await loadKoreanFont(pdfDoc)
  const page = pdfDoc.getPages()[0]

  const jf = parseJumin(patient.ssn ?? '')
  const today = parseDateFields(new Date())
  const manager = caseData.caseManager
  const noiseJobs = ((detail?.workHistory ?? []) as any[]).filter((w: any) => w.noiseExposure)

  // 성명
  drawText(page, patient.name ?? '', 0, 0, font, 9)

  // 주민번호 전체 13자리 개별
  drawText(page, jf['1']  ?? '', 0, 0, font, 9)
  drawText(page, jf['2']  ?? '', 0, 0, font, 9)
  drawText(page, jf['3']  ?? '', 0, 0, font, 9)
  drawText(page, jf['4']  ?? '', 0, 0, font, 9)
  drawText(page, jf['5']  ?? '', 0, 0, font, 9)
  drawText(page, jf['6']  ?? '', 0, 0, font, 9)
  drawText(page, jf['7']  ?? '', 0, 0, font, 9)
  drawText(page, jf['8']  ?? '', 0, 0, font, 9)
  drawText(page, jf['9']  ?? '', 0, 0, font, 9)
  drawText(page, jf['10'] ?? '', 0, 0, font, 9)
  drawText(page, jf['11'] ?? '', 0, 0, font, 9)
  drawText(page, jf['12'] ?? '', 0, 0, font, 9)
  drawText(page, jf['13'] ?? '', 0, 0, font, 9)

  // 주소
  drawText(page, patient.address ?? '', 0, 0, font, 8)

  // 연락처
  drawText(page, patient.phone ?? '', 0, 0, font, 9)

  // 직종 (workHistory 마지막 항목)
  const allJobs = (detail?.workHistory ?? []) as any[]
  const lastJob = allJobs[allJobs.length - 1]
  drawText(page, lastJob?.jobType ?? '', 0, 0, font, 8)

  // 재직/퇴직 체크 (기본: 퇴직)
  drawText(page, '', 0, 0, font, 9)  // 재직 체크 (빈칸)
  drawText(page, '√', 0, 0, font, 9) // 퇴직 체크 (기본값)

  // 사업장명
  drawText(page, caseData.workplaceName ?? '', 0, 0, font, 8)

  // 오늘 날짜
  drawText(page, today.연도, 0, 0, font, 9)
  drawText(page, today.월자, 0, 0, font, 9)
  drawText(page, today.일자, 0, 0, font, 9)

  // 청구인
  drawText(page, patient.name ?? '', 0, 0, font, 9)
  drawText(page, patient.phone ?? '', 0, 0, font, 9)

  // 대리인
  drawText(page, manager?.name ?? '', 0, 0, font, 9)
  drawText(page, manager?.officeTel ?? '', 0, 0, font, 9)

  // 관할공단
  drawText(page, caseData.kwcOfficeName ?? '', 0, 0, font, 9)

  // 소음사업장 직력 (최대 5행)
  const rowCount = Math.min(noiseJobs.length, 5)
  for (let i = 0; i < 5; i++) {
    const job = noiseJobs[i]
    if (job) {
      const period = `${job.startYear}.${String(job.startMonth).padStart(2,'0')}~${job.endYear}.${String(job.endMonth).padStart(2,'0')}`
      drawText(page, job.company ?? '', 0, 0, font, 7)
      drawText(page, period, 0, 0, font, 7)
      drawText(page, job.jobType ?? '', 0, 0, font, 7)
    }
  }

  // 별지사용 체크 (noiseJobs 5개 초과 시 자동 체크)
  drawText(page, noiseJobs.length > 5 ? '√' : '', 0, 0, font, 9)

  const pdfBytes = await pdfDoc.save()
  pdfBytes_result = pdfBytes
  break
}
