## STEP 2: 이유서 내용 저장/조회 API 추가

파일: app/api/objection/cases/[id]/reason/route.ts (신규 생성)

```typescript
import { NextRequest, NextResponse } from 'next/server'
import { auth } from '@/auth'
import { prisma } from '@/lib/prisma'

// GET: 이유서 내용 조회
export async function GET(
  req: NextRequest,
  { params }: { params: { id: string } }
) {
  const session = await auth()
  if (!session) return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })

  const objCase = await prisma.objectionCase.findUnique({
    where: { id: params.id },
    select: { reasonContent: true },
  })

  return NextResponse.json({ reasonContent: objCase?.reasonContent ?? '' })
}

// PATCH: 이유서 내용 저장
export async function PATCH(
  req: NextRequest,
  { params }: { params: { id: string } }
) {
  const session = await auth()
  if (!session) return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })

  const { reasonContent } = await req.json()

  const updated = await prisma.objectionCase.update({
    where: { id: params.id },
    data: { reasonContent },
  })

  return NextResponse.json({ ok: true, reasonContent: updated.reasonContent })
}
```

## STEP 3: PDF 생성 API 추가

파일: app/api/objection/cases/[id]/generate-reason/route.ts (신규 생성)

```typescript
import { NextRequest, NextResponse } from 'next/server'
import { auth } from '@/auth'
import { prisma } from '@/lib/prisma'
import { PDFDocument, rgb } from 'pdf-lib'
import fontkit from '@pdf-lib/fontkit'
import fs from 'fs'
import path from 'path'

// A4 기준 상수
const PAGE_WIDTH = 595.28   // pt
const PAGE_HEIGHT = 841.89  // pt
const MARGIN = 60
const LINE_HEIGHT = 22

export async function POST(
  req: NextRequest,
  { params }: { params: { id: string } }
) {
  const session = await auth()
  if (!session) return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })

  const { docType } = await req.json()
  // docType: 'reason'(이유서) | 'opinion'(의견서)

  // 사건 정보 조회
  const objCase = await prisma.objectionCase.findUnique({
    where: { id: params.id },
    select: {
      patientName: true,
      caseType: true,
      tfName: true,
      decisionDate: true,
      examClaimDate: true,
      reExamClaimDate: true,
      reasonContent: true,
      manager: { select: { name: true } },
    },
  })

  if (!objCase) {
    return NextResponse.json({ error: '사건을 찾을 수 없습니다' }, { status: 404 })
  }

  // PDF 생성
  const pdfDoc = await PDFDocument.create()
  pdfDoc.registerFontkit(fontkit)

  // NotoSansKR 폰트 로드 (기존 프로젝트 폰트 경로 사용)
  const fontPath = path.join(process.cwd(), 'public', 'fonts', 'NotoSansKR-Regular.ttf')
  const fontBytes = fs.readFileSync(fontPath)
  const font = await pdfDoc.embedFont(fontBytes)

  const boldFontPath = path.join(process.cwd(), 'public', 'fonts', 'NotoSansKR-Bold.ttf')
  const boldFontBytes = fs.readFileSync(boldFontPath)
  const boldFont = await pdfDoc.embedFont(boldFontBytes)

  const page = pdfDoc.addPage([PAGE_WIDTH, PAGE_HEIGHT])
  let y = PAGE_HEIGHT - MARGIN

  const docTitle = docType === 'reason' ? '이    유    서' : '의    견    서'

  // 제목
  page.drawText(docTitle, {
    x: PAGE_WIDTH / 2 - boldFont.widthOfTextAtSize(docTitle, 18) / 2,
    y,
    size: 18,
    font: boldFont,
    color: rgb(0, 0, 0),
  })
  y -= LINE_HEIGHT * 2.5

  // 구분선
  page.drawLine({
    start: { x: MARGIN, y },
    end: { x: PAGE_WIDTH - MARGIN, y },
    thickness: 1,
    color: rgb(0, 0, 0),
  })
  y -= LINE_HEIGHT * 1.5

  // 사건 정보 헤더
  const headerItems = [
    { label: '재해자', value: objCase.patientName ?? '' },
    { label: '사건분류', value: objCase.caseType ?? '' },
    { label: '담당 TF', value: objCase.tfName ?? '' },
    { label: '처분일', value: objCase.decisionDate
        ? new Date(objCase.decisionDate).toLocaleDateString('ko-KR')
        : '' },
    { label: '담당 노무사', value: objCase.manager?.name ?? '' },
  ]

  for (const item of headerItems) {
    page.drawText(`${item.label}: ${item.value}`, {
      x: MARGIN,
      y,
      size: 11,
      font,
      color: rgb(0, 0, 0),
    })
    y -= LINE_HEIGHT
  }

  y -= LINE_HEIGHT

  // 구분선
  page.drawLine({
    start: { x: MARGIN, y },
    end: { x: PAGE_WIDTH - MARGIN, y },
    thickness: 0.5,
    color: rgb(0.5, 0.5, 0.5),
  })
  y -= LINE_HEIGHT * 1.5

  // 이유서 본문
  page.drawText('【 이    유 】', {
    x: MARGIN,
    y,
    size: 13,
    font: boldFont,
    color: rgb(0, 0, 0),
  })
  y -= LINE_HEIGHT * 1.5

  const content = objCase.reasonContent ?? '(내용 없음)'
  const maxWidth = PAGE_WIDTH - MARGIN * 2
  const fontSize = 11

  // 줄바꿈 처리 (한 줄에 들어가는 글자 수 기준으로 분할)
  const lines = content.split('\n')
  for (const line of lines) {
    // 빈 줄 처리
    if (!line.trim()) {
      y -= LINE_HEIGHT
      continue
    }
    // 긴 줄 자동 줄바꿈
    let remaining = line
    while (remaining.length > 0) {
      let chunk = remaining
      while (font.widthOfTextAtSize(chunk, fontSize) > maxWidth && chunk.length > 1) {
        chunk = chunk.slice(0, -1)
      }
      page.drawText(chunk, {
        x: MARGIN,
        y,
        size: fontSize,
        font,
        color: rgb(0, 0, 0),
      })
      y -= LINE_HEIGHT
      remaining = remaining.slice(chunk.length)

      // 페이지 넘침 처리 (기본 대응 — 여백 부족 시 새 페이지)
      if (y < MARGIN + LINE_HEIGHT * 3) {
        const newPage = pdfDoc.addPage([PAGE_WIDTH, PAGE_HEIGHT])
        y = PAGE_HEIGHT - MARGIN
        // 이후 내용은 새 페이지에서 계속
      }
    }
  }

  // 서명란
  y -= LINE_HEIGHT * 2
  const today = new Date().toLocaleDateString('ko-KR')
  page.drawText(`작성일: ${today}`, {
    x: PAGE_WIDTH - MARGIN - font.widthOfTextAtSize(`작성일: ${today}`, 11),
    y,
    size: 11,
    font,
    color: rgb(0, 0, 0),
  })
  y -= LINE_HEIGHT * 1.5
  page.drawText(`노무법인 더보상  ${objCase.manager?.name ?? '담당노무사'}  (인)`, {
    x: PAGE_WIDTH - MARGIN - font.widthOfTextAtSize(`노무법인 더보상  ${objCase.manager?.name ?? '담당노무사'}  (인)`, 11),
    y,
    size: 11,
    font,
    color: rgb(0, 0, 0),
  })

  const pdfBytes = await pdfDoc.save()

  const fileName = `${docType === 'reason' ? '이유서' : '의견서'}_${objCase.patientName}_${today.replace(/\./g, '')}.pdf`

  return new NextResponse(Buffer.from(pdfBytes), {
    status: 200,
    headers: {
      'Content-Type': 'application/pdf',
      'Content-Disposition': `attachment; filename*=UTF-8''${encodeURIComponent(fileName)}`,
    },
  })
}
```

### 4-3. PDF 생성 핸들러 교체
기존 handleGenerate에서 토스트만 띄우는 부분을 아래로 교체:

```typescript
const handleGenerate = async () => {
  if (!selectedCase) {
    alert('사건을 먼저 선택해주세요.')
    return
  }

  // 서식 유형 판단 (selectedForm 또는 selectedTemplate 등 기존 변수명 확인 후 사용)
  // 이유서 → 'reason', 의견서 → 'opinion', 나머지 → 추후 구현
  const isReason = selectedForm === '이유서'    // 기존 변수명에 맞게 수정
  const isOpinion = selectedForm === '의견서'   // 기존 변수명에 맞게 수정

  if (!isReason && !isOpinion) {
    alert('이유서 또는 의견서를 선택해주세요. 나머지 서식은 준비 중입니다.')
    return
  }

  try {
    const res = await fetch(`/api/objection/cases/${selectedCase.id}/generate-reason`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ docType: isReason ? 'reason' : 'opinion' }),
    })

    if (!res.ok) {
      alert('PDF 생성 실패. 다시 시도해주세요.')
      return
    }

    // PDF 다운로드
    const blob = await res.blob()
    const url = URL.createObjectURL(blob)
    const a = document.createElement('a')
    a.href = url
    a.download = `이유서_${selectedCase.patientName}.pdf`
    a.click()
    URL.revokeObjectURL(url)
  } catch (e) {
    alert('PDF 생성 중 오류가 발생했습니다.')
  }
}
```
